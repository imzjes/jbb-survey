# Extra requirements for the local-inference path used in Phase 2 / 3
# (PerplexityFilter and EraseAndCheck rely on auxiliary local models).
# Install on a CUDA host (e.g. Colab Pro) with:
#   uv pip install --python <venv>/bin/python -r requirements.txt -r requirements-vllm.txt
#
# Pins are tightened because newer transformers (>=4.50) removes the
# fallback to torch's private `_register_pytree_node`, which breaks against
# any torch <2.3. The pinned window below is a tested mutually-compatible
# GPU stack.
torch>=2.4.0,<2.6.0
transformers>=4.46.0,<4.50.0
numpy>=1.24.0,<2.0.0
# Pin vllm directly (skip the jbb[vllm] extra) so we control the version.
# vllm 0.3.3 (which jbb[vllm] would pull) is ABI-incompatible with torch 2.5.x.
# vllm 0.6.6 was built against torch 2.5.1 and works with our pin window.
vllm==0.6.6
jailbreakbench>=1.0.0
